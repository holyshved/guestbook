CREATE TABLE users (
   id int(11) unsigned NOT NULL auto_increment,
   password text NOT NULL,
   name text NOT NULL,
   email text NOT NULL,
   avatar varchar(255) NOT NULL,
   PRIMARY KEY (id)
);

CREATE TABLE messages (
   id int(11) unsigned NOT NULL auto_increment,
   userid int(11) NOT NULL,
   date datetime NOT NULL,
   text text NOT NULL,
   PRIMARY KEY (id)
);
